Global Material Change : Sales Model¶

This demo is to show how to build a machine learning model in Python (sklearn) from beginning to end.¶

This is a binary classification model, to predict if a company will have more than 15% growth of sales in next 12 months.¶

The modelled population includes 24 countries. The dataset has been simplified for demo purpose.¶

  • Data: Estimated Annual Sales
  • Timeframe: observation as of Jan. 2014 (development sample)
  • Model Type: Classification
  • Algorithm: Random Forest (optimized hyperparameter)
  • Filter : Estimated sales figure only
  • Target Variable: Growth (vs. deterioration)

def preprocess(df):
    
    # Record values
    mktb_mapping = {
               np.nan: 0,
               'M': 1}
    
    dlst_mapping = {'N': 0,
               'Y': 1}
    
    df['dlst_ind_c'] = df['dlst_ind_c'].map(dlst_mapping)
    df['dlst_ind_p'] = df['dlst_ind_p'].map(dlst_mapping)
    
    df['mktb_ind_p'] = df['mktb_ind_p'].map(mktb_mapping)
    pd.unique(df.mktb_ind_p)
    
    #Create dummy variables
    df['SIC2D_P2'] = df['SIC2D_P'].astype(str)
    B = pd.get_dummies(df[['imp_exp_agt_code_ind_p','SIC2D_P2']])
    df = pd.concat([df,B],axis = 1)
    
    #Fill missing values
    df['glbl_hier_code_p'].fillna(99,inplace=True)
    pd.unique(df['glbl_hier_code_p'])
    
    #Drop unused variables
    df = df.drop([
                  'imp_exp_agt_code_ind_p','sales_grp_p','sales_grp_c','mfd_ind_p','SIC2D_P','SIC2D_P2'\
                  ],1)
    #Double check if there is any non-numeric variables         
    for i,j in zip(df.columns,df.dtypes):
        if j!='int64' and j!='float64':
            print i ,j
            
    #Double check if there is any missing values   
    def check_type(in1):
        A= in1.isnull().sum()
        if len(A[A>0].values)>0: 
            print A[A>0]
         
    check_type(df)
    return df

def train(X_train, X_test, y_train, y_test):
    
    #Define the algorithm, specify the hyperparameters
    forest = RandomForestClassifier(criterion='gini',
                                    n_estimators=1000, 
                                    max_depth = 25,
                                    min_samples_leaf=25,
                                    max_features=50,
                                    n_jobs=-1
                                    )
    # Train the model with training dataset
    forest.fit(X_train, y_train,sample_weight = X_train.weight_sales.values)
    
    return forest

def assessment2(proba,y_test,num_bins):

        D = pd.DataFrame(proba[:, 1],columns = ['y_prob'])
        E = pd.DataFrame(y_test.values,columns=['y_test'])
        Results = pd.concat([D,E],1)
        del D,E
        bins = np.unique(algos.quantile(Results['y_prob'], np.linspace(0, 1, num_bins)))
        Results['decile']  = pd.tools.tile._bins_to_cuts(Results['y_prob'], bins,labels=np.arange( 0,len(bins)-1,1)[::-1], include_lowest=True)
        
        group = Results.groupby(['decile','y_test'])
        Results2 = group.y_test.count().unstack().fillna(0).sort_index(ascending=False)
        Results2.columns = ['Negative','Positive']
        Total_neg,Total_pos = Results2.sum()[0],Results2.sum()[1]
        Results2['Neg_perc'] = Results2['Negative']/Total_neg
        Results2['Pos_perc'] = Results2['Positive']/Total_pos
        Results2['Total_perc'] = (Results2['Negative']+Results2['Positive'])/(Total_neg+Total_pos)
        Results2['Cul_neg_perc'] = Results2['Neg_perc'].cumsum(axis=0)
        Results2['Cul_pos_perc'] = Results2['Pos_perc'].cumsum(axis=0)
        Results2['Cul_tot_perc'] = Results2['Total_perc'].cumsum(axis=0)
        Results2['Lift_perc'] = (Results2['Cul_pos_perc']+0.0001)/(Results2['Cul_neg_perc']+0.0001)
        Results2['ratio'] =  (Results2['Pos_perc']+0.0001)/(Results2['Neg_perc']+0.0001)
        # Gain chart: Cumulative %pos and Cummulative %Population
        # Lift chart: Decile & model
        KS_value = max(Results2['Cul_pos_perc']-Results2['Cul_neg_perc'])
        print 'KS value :{0}'.format(KS_value) 
        
        num_dec = sum(x>y for x, y in zip(Results2['ratio'], Results2['ratio'][1:]))
        print "{0} out of {1} is strictly decreasing".format(num_dec,len(Results2['ratio'])-1)     
        
        return KS_value,Results2

def rf_importance(forest):
    feat_labels = X_train.columns
    importances = forest.feature_importances_
    indices = np.argsort(importances)[::-1]
    
    importance_list = []
    for f in range(X_train.shape[1]):
        if f < 100:
            print("%2d) %-*s %f" % (f + 1, 30, 
                               feat_labels[indices[f]],
                               importances[indices[f]]))
        importance_list.append((f,feat_labels[indices[f]],importances[indices[f]]))
    return importance_list

Out-of-time validation¶

By default Python load the whole file into memory. When dealing with large files, it is recommended to split file into chunks. Score each chunk sequentially, and only keep the output target variable.

In [46]:
import pandas as pd
reader = pd.read_csv('/data2/GMC/sales_model/SALES_DEVELOPMENT_cris.csv', iterator=True)

loop = True
chunkSize = 500000
y_set_oot = pd.Series([])
proba_pred_set_oot = pd.DataFrame([])
X_weight_set_oot = pd.Series([])
i= 1
while loop:
    start_time = time.time()
    try:
        chunk = reader.get_chunk(chunkSize)
        chunk = chunk.set_index(chunk.duns)
        chunk2 = preprocess(chunk)
        y=chunk2['increase_sales']
        X=chunk2.drop(['increase_sales','nochange_sales','decrease_sales','sales_pct_ALL','ann_sls_ind_c','ann_sls_ind_p',\
                 'ann_sls_us_dolr_amt_c','duns','hq_par_ctry_cd_c','dlst_ind_p','dlst_ind_c'],1)
        misscolumn = (set(X_train.columns.values)-set(X.columns.values)) 
        for a in misscolumn:
            X[a]=0
        proba_pred = forest.predict_proba(X)
        proba_pred = pd.DataFrame(proba_pred).set_index(chunk.duns)
        proba_pred_set_oot=pd.concat([proba_pred_set_oot,proba_pred], axis=0)
        
        X_weight_set_oot = pd.concat([X_weight_set_oot,X['weight_sales']])
        y_set_oot = pd.concat([y_set_oot,y])
        print("---Chunk %s :  %s seconds ---" % (i,time.time() - start_time))
        i= i+1
        
    except StopIteration:
        loop = False
        print "Iteration is stopped."
